            USER {
                string uid PK
                string displayName
                string photoURL
                string bio
            }
            
            GALLERY {
                string id PK
                string ownerId FK
                string name
                string style
            }
            
            ARTWORK {
                string id PK
                string galleryId FK
                string imageUrl
            }
            
            CHAT {
                string id PK
                string[] participants
            }
            
            NOTIFICATION {
                string id PK
                string ownerId FK
                boolean read
            }
